area.html: computes area of circle until no input 

<script>
    var rad, area;
    rad = prompt("Enter radius")
    while (rad != null) {
    	area = 3.14 * rad * rad
        alert("radius = " + rad + ", area = " + area)
        document.write("<br> radius = " + rad + ", area = " + area)
    	rad = prompt("Enter radius")
    }
</script>
